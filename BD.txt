/*CRIANDO A PROCEDURE QUE CONCATENA O PREÇO DO LIVRO */

DELIMITER $$

CREATE PROCEDURE VERPRECO(vlivro int)
BEGIN

SELECT CONCAT(' O PREÇO' , NOME_LIVRO, 'É' , PRECO) AS PREÇO
FROM LIVROS
WHERE ID_LIVRO = vlivro;

END $$
