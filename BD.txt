/*CRIANDO O BANCODE DADOS LIVRARIA */
CREATE DATABASE PLIVRARIA;

/*USANDO O BANCO DE DADOS LIVRARIA */
USE PLIVRARIA;

/*CRIANDO AS TABELAS DO BANCO DE DADOS LIVRARIA */

ID_LIVRO INT,
ID_EDITORA INT,
NOME_LIVRO VARCHAR(50),

);

CREATE TABLE EDITORA(
ID_EDITORA INT,
NOME_EDITORA VARCHAR(50)
);
